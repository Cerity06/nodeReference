1. Modeling user using a schema
2. Creating new users with a signup request
3. Managing password

Crypt with 'bcrypt' => npm i bcryptjs

```ts
memberSchema.pre('save', async function (next) {
  // only runs if password was actually modified
  if (!this.isModified('password')) {
    return next();
  }

  this.password = await bcrypt.hash(this.password, 12);

  // delete the password confirmed
  this.passwordConfirmed = undefined;
});
```

## npm for security purposes (limiter and helmet (pack of security middleware))

```ts
const rateLimit = requite('express-rate-limit');
const helmet = requite('helmet');

// Set security headers
app.use(helmet());

// Limit requests from the client
const limiter = ratelimit({
  max: 1000,
  windowMs: 60 * 60 * 1000,
  message: 'Too many requests from client, please try again in an hour!',
});

app.use('/api', limiter);

// if body greater than 10kb, it will not be allowed
app.use(express.json({ limit: '10kb' }));
```

## Data sanitization

- Clean malicious code: XSS and noSQL query injection attacks

- npm i express-mongo-sanitize
- npm i xss-clean

```ts
const mongoSanitize = require('express-mongo-sanitize');
const xss = require('xss-clean');
const hpp = require('hpp');

app.use(mongoSanitize()); // avoid using mongo API in body by malicious users
app.use(xss()); // convert any html code injected to plain text

// remove duplicate fields and clear up search query except if in whitelist
app.use(
  hpp({
    ['duration', 'difficulty', 'price']
  })
);
```
